    module Prism::Action::Auth

    Overview

    Auth module for Prism::Action.

    struct MyAction < Prism::Action
  include Auth

  def call
    if auth?    # Check if auth object exists in current request and answers #auth
      auth.user # It will return a User instance
    else
      halt!(401)
    end
  end
end
